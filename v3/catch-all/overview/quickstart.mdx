This guide helps you make your first API call to transform natural language
questions into structured, validated records with source citations.

## Before you start

Before you begin, make sure you meet these prerequisites:

- An API key (obtain one through
  [Book a demo](https://www.newscatcherapi.com/book-a-demo) page)
- Basic understanding of REST APIs
- Your preferred programming language and HTTP client
- Basic knowledge of JSON data format

## Steps

<Steps>
  <Step title="Set up environment">

    Install the CatchAll client library for your programming language:

    <CodeGroup>
    ```bash curl
    # cURL is typically included in your system
    # To check, open the terminal and type the following:
    curl --version
    ```

    ```bash Python
    pip install newscatcher-catchall-sdk
    ```

    ```bash TypeScript
    npm install newscatcher-catchall-sdk
    ```

    ```bash Java (Gradle)
    dependencies {
        implementation 'com.newscatcherapi:newscatcher-catchall-sdk:0.1.0'
    }
    ```

    ```bash Java (Maven)
    <dependency>
        <groupId>com.newscatcherapi</groupId>
        <artifactId>newscatcher-catchall-sdk</artifactId>
        <version>0.1.0</version>
    </dependency>
    ```
    </CodeGroup>

  </Step>

  <Step title="Create job">

    Submit a natural language query to create a new processing job:

    <CodeGroup>
    ```bash curl
    curl -X POST https://catchall.newscatcherapi.com/catchAll/submit \
      -H "x-api-key: YOUR_API_KEY_HERE" \
      -H "Content-Type: application/json" \
      -d '{
        "query": "AI company acquisitions",
        "context": "Focus on deal size and acquiring company details"
      }'
    ```

    ```json JSON
    {
      "query": "AI company acquisitions",
      "context": "Focus on deal size and acquiring company details",
    }
    ```

    ```python Python
    from newscatcher_catchall import CatchAllApi

    # Initialize the client
    client = CatchAllApi(api_key="YOUR_API_KEY_HERE")

    # Create a job
    job = client.jobs.create_job(
        query="AI company acquisitions",
        context="Focus on deal size and acquiring company details"
    )

    print(f"Job created: {job.job_id}")
    ```

    ```typescript TypeScript
    import { CatchAllApiClient } from "newscatcher-catchall-sdk";

    // Initialize the client
    const client = new CatchAllApiClient({ apiKey: "YOUR_API_KEY_HERE" });

    // Create a job
    const job = await client.jobs.createJob({
      query: "AI company acquisitions",
      context: "Focus on deal size and acquiring company details"
    });

    console.log(`Job created: ${job.jobId}`);
    ```

    ```java Java
    import com.newscatcher.catchall.CatchAllApi;
    import com.newscatcher.catchall.resources.jobs.requests.SubmitRequestDto;

    // Initialize the client
    CatchAllApi client = CatchAllApi.builder()
        .apiKey("YOUR_API_KEY_HERE")
        .build();

    // Create a job
    var job = client.jobs().createJob(
        SubmitRequestDto.builder()
            .query("AI company acquisitions")
            .context("Focus on deal size and acquiring company details")
            .build()
    );

    System.out.println("Job created: " + job.getJobId());
    ```
    </CodeGroup>

    <Note>
      Replace `YOUR_API_KEY_HERE` with your actual API key.
    </Note>

    You receive a response with a job ID:

    ```json
    {
      "job_id": "af7a26d6-cf0b-458c-a6ed-4b6318c74da3"
    }
    ```

    Use this `job_id` to check the status and retrieve results.

  </Step>

  <Step title="Check job status">

    Wait 1-2 minutes, then check the processing status. Poll this endpoint every 30-60 seconds until the status is `job_completed`:

    <CodeGroup>
    ```bash curl
    curl -X GET https://catchall.newscatcherapi.com/catchAll/status/af7a26d6-cf0b-458c-a6ed-4b6318c74da3 \
      -H "x-api-key: YOUR_API_KEY_HERE"
    ```

    ```python Python
    from newscatcher_catchall import CatchAllApi
    import time

    client = CatchAllApi(api_key="YOUR_API_KEY_HERE")
    job_id = "af7a26d6-cf0b-458c-a6ed-4b6318c74da3"

    # Poll until job completes
    while True:
        status = client.jobs.get_job_status(job_id)

        print(f"Current status: {status.status}")

        # Check if job is completed by examining steps
        completed_step = next((s for s in status.steps if s.status == 'completed' and s.completed), None)
        failed_step = next((s for s in status.steps if s.status == 'failed' and s.completed), None)

        if completed_step:
            print("Job completed!")
            break
        elif failed_step:
            print("Job failed!")
            break

        time.sleep(60)  # Wait 60 seconds before next check
    ```

    ```typescript TypeScript
    import { CatchAllApiClient } from "newscatcher-catchall-sdk";

    const client = new CatchAllApiClient({ apiKey: "YOUR_API_KEY_HERE" });
    const jobId = "af7a26d6-cf0b-458c-a6ed-4b6318c74da3";

    // Poll until job completes
    while (true) {
      const status = await client.jobs.getJobStatus(jobId);

      console.log(`Current status: ${status.status}`);

      // Check if job is completed by examining steps
      const completedStep = status.steps.find(s => s.status === 'completed' && s.completed);
      const failedStep = status.steps.find(s => s.status === 'failed' && s.completed);

      if (completedStep) {
        console.log("Job completed!");
        break;
      } else if (failedStep) {
        console.log("Job failed!");
        break;
      }

      await new Promise(resolve => setTimeout(resolve, 60000)); // Wait 60 seconds
    }
    ```

    ```java Java
    import com.newscatcher.catchall.CatchAllApi;
    import com.newscatcher.catchall.types.StatusResponseDto;

    CatchAllApi client = CatchAllApi.builder()
        .apiKey("YOUR_API_KEY_HERE")
        .build();

    String jobId = "af7a26d6-cf0b-458c-a6ed-4b6318c74da3";

    // Poll until job completes
    while (true) {
        StatusResponseDto status = client.jobs().getJobStatus(jobId);

        System.out.println("Current status: " + status.getStatus());

        // Check if job is completed by examining steps
        var completedStep = status.getSteps().stream()
            .filter(s -> "completed".equals(s.getStatus()) && s.getCompleted())
            .findFirst();
        var failedStep = status.getSteps().stream()
            .filter(s -> "failed".equals(s.getStatus()) && s.getCompleted())
            .findFirst();

        if (completedStep.isPresent()) {
            System.out.println("Job completed!");
            break;
        } else if (failedStep.isPresent()) {
            System.out.println("Job failed!");
            break;
        }

        Thread.sleep(60000); // Wait 60 seconds
    }
    ```
    </CodeGroup>

    either `completed` (order 6) or `failed` (order 7). Processing typically
    takes 10-15 minutes.

  </Step>

  <Step title="Retrieve results">

    Once the status is `job_completed`, retrieve the structured records:

    <CodeGroup>
    ```bash curl
    curl -X GET https://catchall.newscatcherapi.com/catchAll/pull/af7a26d6-cf0b-458c-a6ed-4b6318c74da3 \
      -H "x-api-key: YOUR_API_KEY_HERE"
    ```

    ```python Python
    from newscatcher_catchall import CatchAllApi

    client = CatchAllApi(api_key="YOUR_API_KEY_HERE")
    job_id = "af7a26d6-cf0b-458c-a6ed-4b6318c74da3"

    # Retrieve results
    results = client.jobs.get_job_results(job_id)

    print(f"Query: {results.query}")
    print(f"Valid records: {results.valid_records}")
    print(f"Candidate records: {results.candidate_records}")
    print(f"Duration: {results.duration}")

    # Display first record
    if results.all_records:
        record = results.all_records[0]
        print(f"\nFirst record: {record.record_title}")
        print("\nExtracted fields:")
        for key, value in record.enrichment.items():
            print(f"  {key}: {value}")
    ```

    ```typescript TypeScript
    import { CatchAllApiClient } from "newscatcher-catchall-sdk";

    const client = new CatchAllApiClient({ apiKey: "YOUR_API_KEY_HERE" });
    const jobId = "af7a26d6-cf0b-458c-a6ed-4b6318c74da3";

    // Retrieve results
    const results = await client.jobs.getJobResults(jobId);

    console.log(`Query: ${results.query}`);
    console.log(`Valid records: ${results.validRecords}`);
    console.log(`Candidate records: ${results.candidateRecords}`);
    console.log(`Duration: ${results.duration}`);

    // Display first record
    if (results.allRecords.length > 0) {
      const record = results.allRecords[0];
      console.log(`\nFirst record: ${record.recordTitle}`);
      console.log("\nExtracted fields:");
      Object.entries(record.enrichment).forEach(([key, value]) => {
        console.log(`  ${key}: ${value}`);
      });
    }
    ```

    ```java Java
    import com.newscatcher.catchall.CatchAllApi;
    import com.newscatcher.catchall.types.PullJobResponseDto;

    CatchAllApi client = CatchAllApi.builder()
        .apiKey("YOUR_API_KEY_HERE")
        .build();

    String jobId = "af7a26d6-cf0b-458c-a6ed-4b6318c74da3";

    // Retrieve results
    PullJobResponseDto results = client.jobs().getJobResults(jobId);

    System.out.println("Query: " + results.getQuery());
    System.out.println("Valid records: " + results.getValidRecords());
    System.out.println("Candidate records: " + results.getCandidateRecords());
    System.out.println("Duration: " + results.getDuration());

    // Display first record
    if (!results.getAllRecords().isEmpty()) {
        var record = results.getAllRecords().get(0);
        System.out.println("\nFirst record: " + record.getRecordTitle());
        System.out.println("\nExtracted fields:");
        record.getEnrichment().forEach((key, value) -> {
            System.out.println("  " + key + ": " + value);
        });
    }
    ```
    </CodeGroup>
